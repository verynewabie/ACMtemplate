//x的欧拉函数是指1~x中与x互质的数的个数
//求单个数的欧拉函数
int euler(int x){
    int res=x;
    for(int i=2;i<=x/i;i++){
        if(x%i==0){
            res=res/i*(i-1);
            while(x%i==0){
                x/=i;
            }
        }
    }
    if(x>1) res=res/x*(x-1);
    return res;
}

//筛法求欧拉函数
void euler(int tot){
	v[1]=1;
	for(int i=2;i<=tot;i++){
		if(!st[i]){  
			primes[cnt++]=i;
			v[i]=i-1;
		}
		for(int j=0;primes[j]<=tot/i;j++){
			int t=primes[j]*i;
			st[t]=true;
			if(i%primes[j]==0){
				v[t]=v[i]*primes[j];
				break;
			}
			v[t]=v[i]*(primes[j]-1);
		}
	}
}
