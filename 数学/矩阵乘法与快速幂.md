struct martix{
	int n,m;
	int w[N][N]={0};
};
martix operator*(martix a,martix b){
	martix ans;
	ans.n=a.n,ans.m=b.m;
	for(int i=1;i<=a.n;i++)
		for(int j=1;j<=b.m;j++)
			for(int k=1;k<=a.m;k++)
				ans.w[i][j]=(ans.w[i][j]+a.w[i][k]*b.w[k][j]);
	return ans;
}
martix qmi(martix a,ll b){
	martix ans;
	ans.n=ans.m=a.n;
	for(int i=1;i<=ans.m;i++) ans.w[i][i]=1;
	while(b){
		if(b&1) ans=ans*a;
		a=a*a;
		b>>=1;
	}
	return ans;
}
