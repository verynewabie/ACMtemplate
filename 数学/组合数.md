## 卢卡斯定理

```cpp
int n,p;
int qmi(int a,int k){
    int res=1;
    while(k){
        if(k&1)res=(LL)res*a%p;
        a=(LL)a*a%p;
        k>>=1;
    }
    return res;
}
int C(int a,int b){
    if(b>a) return 0;
    int res=1;
    for(int i=b,j=a-b+1;i>=1;i--,j++){
        res=(LL)res*qmi(i,p-2)%p;
        res=(LL)res*j%p;
    }
    return res;
}
int lucas(LL a,LL b){
    if(a<p&&b<p)    return C(a,b)%p;
    else return (LL)C(a%p,b%p)*lucas(a/p,b/p)%p;
}
int main(){//求C(a,b)%p的值
    cin>>n;
    while(n--){
        LL a,b;
        cin>>a>>b>>p;
        cout<<lucas(a,b)<<endl;
    }
    return 0;
}
```
