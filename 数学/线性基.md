struct linebasis{
    ll a[64],newa[64];
    bool flag;
    int tot;
    void init(){
        memset(a,0,sizeof a);
        memset(newa,0,sizeof newa);
        tot=flag=0;
    }
    void insert(ll x){
        for(int i=62;~i;i--)
            if(x>>i&1)
                if(!a[i]){a[i]=x;return ;}
                else x^=a[i];     
        flag=1;
    }
    bool check(ll x){
        for(int i=62;~i;i--)
            if(x>>i&1)
                if(!a[i]) return false;
                else x^=a[i];
        return true;
    }
    ll qmax(){
        ll res=0;
        for(int i=62;~i;i--)
            res=max(res,res^=a[i]);
        return res;
    }
    ll qmin(){
        if(flag) return 0;
        for(int i=0;i<=62;i++)
            if(a[i])
                return a[i];
    }
    void rebuild(){//查询k小值之前要rebuild
        for(int i=0;i<=62;i++)
            for(int j=0;j<i;j++)
                if(a[i]>>j&1)
                    a[i]^=a[j];
        for(int i=0;i<=62;i++)
            if(a[i])
                newa[cnt++]=a[i];
	}
    ll kth(ll k){//第k小
		if(flag) k--;
		if(k==0) return 0;
		if(k>=(1ll<<tot)) return -1;
		ll res=0;
        for(int i=0;i<=62;i++)
            if(k>>i&1) 
                res^=newa[i];
		return res;
	}
};
