#include<bits/stdc++.h>
using namespace std;
using ll=long long;
using pii=pair<int,int>;
const int N=1e4+10;
int n,m;
bool st[N];//st[i]表示i是否被删掉
vector<pii> to[N];
int p[N],q[N];
int getsz(int u,int fa){
    if(st[u]) return 0;
    int res=1;
    for(auto [j,w]:to[u]){
        if(j==fa) continue;
        res+=getsz(j,u);
    }
    return res;
}
int getforce(int u,int fa,int tot,int& force){
    if(st[u]) return 0;
    int sum=1,ma=0;
    for(auto [j,w]:to[u]){
        if(j==fa) continue;
        int sz=getforce(j,u,tot,force);
        ma=max(sz,ma);
        sum+=sz;
    }
    ma=max(ma,tot-sum);
    if(ma<=tot/2) force=u;
    return sum;
}
void getdist(int u,int fa,int dist,int& szq){
    if(st[u]) return ;
    q[szq++]=dist;
    for(auto [j,w]:to[u]){
        if(j==fa) continue;
        getdist(j,u,dist+w,szq);
    }
}
int get(int* a,int sz){
    sort(a,a+sz);
    int res=0;
    for(int i=sz-1,j=-1;i>=0;i--){
        while(j+1<i&&a[j+1]+a[i]<=m) j++;
        j=min(j,i-1);
        res+=j+1;
    }
    return res;
}
ll calc(int u){
    if(st[u]) return 0;
    int res=0;
    getforce(u,-1,getsz(u,-1),u);
    st[u]=1;
    int szp=0;
    for(auto [j,w]:to[u]){
        int szq=0;
        getdist(j,-1,w,szq);//得到子树中到重心的距离
        res-=get(q,szq);//减去该子树中多的贡献
        for (int k=0;k<szq;k++){
            if(q[k]<=m) res++;//顺便统计下子树到重心的答案
            p[szp++]=q[k];
        }
    }
    res+=get(p,szp);//算下子树间的答案
    for(auto [j,w]:to[u]) res+=calc(j);
    return res;
}
int main(){
    while(scanf("%d%d",&n,&m),n||m){
        for(int i=0;i<n;i++) to[i].clear();
        memset(st,0,sizeof st);
        for(int i=0;i<n-1;i++){
            int u,v,l;
            scanf("%d%d%d",&u,&v,&l);
            to[u].push_back({v,l});
            to[v].push_back({u,l});
        }
        printf("%lld\n",calc(0));    
    }
    return 0;
}
