//有向图
#include<bits/stdc++.h>
using namespace std;
const int N=1e5+10,M=2e5+10;
int n,m;
int h[N],e[M],ne[M],idx;
bool use[N];
int stk[N],top;
void add(int a,int b){
	e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}
void dfs(int u){
	for(int &i=h[u];~i;){
		if(use[i]){
			i=ne[i];
			continue;
		}
		use[i]=1;
		int j=e[i];
		i=ne[i];
		dfs(j);
		//stk[++top]=;记录边
	}
	stk[++top]=u;
}
int main(){
	memset(h,-1,sizeof h);
	cin>>n>>m;
	while(m--){
		int a,b;
		cin>>a>>b;
		add(a,b);
	}
	//判断是否有解
	dfs(1);//还要判断下从哪个点开始搜
	while(top) cout<<stk[top--]<<" \n"[top==0];//输出点，边同理
	return 0;
}

//无向图
#include<bits/stdc++.h>
using namespace std;
const int M=2e5+10,N=1e5+10;
int m,n,e[M],ne[M],h[N],idx,cnt,din[N],dout[N],ans[N];
bool use[M];
void add(int a,int b){
    e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}
void dfs(int u){
    for(int &i=h[u];~i;){//因为要修改i指向的内容，因此用到了引用
        if(use[i]){
            i=ne[i];//删除这条边
            continue;
        }
        use[i]=1;
        use[i^1]=1;//反向边也要标记
        int t=i/2+1;
        if(i&1) t=-t;//反向边
        int j=e[i];
        i=ne[i];
        dfs(j);
        ans[++cnt]=t;
    }
}
int main(){
    cin>>n>>m;
    memset(h,-1,sizeof h);
    for(int i=1;i<=m;i++){
        int a,b;
        cin>>a>>b;
        add(a,b);
        add(b,a);
        din[b]++,dout[a]++;
    }
    //无向边的度就相当于入度加出度
    for(int i=1;i<=n;i++)
        if(din[i]+dout[i]&1){
            puts("NO");//无解
            return 0;
        }
    dfs(1);
    if(cnt<m){
        puts("NO");//无解
        return 0;
    }
    for(int i=cnt;i;i--){
        cout<<ans[i]<<" ";
    }
    return 0;
}
